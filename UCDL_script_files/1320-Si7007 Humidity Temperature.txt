 
Function1(Pin Configuration){
    PIN(1, PWM, DR, 1.22kHz);
    PIN(2, GND, 0V, Open-Drain(type2));
    PIN(3, VDD, 3.5V, Open-Drain(type2));
    PIN(4, CW, CS, Push-Pull(output));
}


Function2(Data Computation){
    DF(Y1, Humidity, -6+125*X1, %RH);
    DF(Y2, Temperature, -46.85+175.72*X2, Degree);
}

Function3(Read PWM Data){

    CW(4, H), + 500ms;
    DR(1, 1, X1),+100us;
    CW(4, H), + 500ms;
    DR(1, 1, X1),+100us;

}
