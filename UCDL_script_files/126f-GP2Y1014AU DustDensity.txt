 
Function1(Pin Configuration){
    PIN(1, VCC, 5V, Open-Drain(type2));
    PIN(2, GND, 0V, Open-Drain(type2));
    PIN(3, CW, CW, Push-Pull(output));
    PIN(4, GND, 0V, Open-Drain(type2));
    PIN(5, ADC, 5V);
    PIN(6, VCC, 5V, Open-Drain(type2));
}
 
Function2(Data Computation){
    DF(Y1, DustDensity, 0.17*(X1<<8+X2)*5/65535-0.1, mg/m^3);
}

Function3(ADC Sampling){
    CW(3, H), +280us;
    DR(5, ADC, X1, X2), +40us;
    CW(3, L), +9680us;
}
