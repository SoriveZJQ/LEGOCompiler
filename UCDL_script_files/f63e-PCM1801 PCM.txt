 
Function1(Pin Configuration){
    PIN(1, PCM, SCK, 12MHz);
    PIN(2, PCM, BCK, Standard, 24Bits, 32Frame, 8kHz);
    PIN(3, PCM, WS);
    PIN(4, PCM, DR);
    PIN(5, GND, 0V, Open-Drain(type2));
    PIN(6, GND, 0V, Open-Drain(type2));
    PIN(7, VDD, 5V, Open-Drain(type2));
}

Function2(Get PCM Data){
    DR(4, PCM, DIRECT), +1s;
}
