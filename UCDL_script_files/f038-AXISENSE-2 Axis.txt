 
Function1(Pin Configuration){
    PIN(1, VDD, 10V, Open-Drain(type2));
    PIN(2, GND, 0V, Open-Drain(type2));
    PIN(3, ADC, 5V);
    PIN(4, ADC, 5V);
}

Function2(Data Computation){
    DF(Y1, X_Axis, ((5/pow(2,16))*(X1<<8+X2)-2.5)*45, Degree);
    DF(Y2, Y_Axis, ((5/pow(2,16))*(X3<<8+X4)-2.5)*45, Degree);
}

Function3(Read Analog Data){
    DR(3, ADC,X1, X2), +10us;
    DR(4, ADC,X3, X4), +3ms;
}
